Replace elements from array

splice() method

The splice() method changes the contents of an array by replacing existing elements.

The splice() method also cut the elements from array.

# Array.splice(position to delete,items to delete,new_element);

let languages = ['C', 'C++', 'Java', 'JavaScript'];

languages.splice(1, 1, 'Python');

Output : ["C", "Python", "Java", "JavaScript"]

# Array.splice(position to delete,items to delete,new_element_1,new_element_2,new_element_3,...);

languages.splice(2, 1, 'C#', 'Swift', 'Go');

Output : ["C", "Python", "C#", "Swift", "Go", "JavaScript"]

const fruits = ['Apple', 'Orange', 'Mango', 'Banana'];

const replace = fruits.splice(1, 2, 'Cherry', 'Watermelon');

console.log(fruits);

Output : ['Apple', 'Cherry', 'Watermelon', 'Banana']

console.log(replace);

Output : ['Orange', 'Mango']
