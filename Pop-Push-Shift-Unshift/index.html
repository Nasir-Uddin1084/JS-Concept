Pop Push Shift Unshift

# pop() : Remove an item from the end of an array.

let brothers = ['Nasir', 'Antor', 'Apon'];
brothers.pop(); // ['Nasir', 'Antor']

# push() : Add items to the end of an array.

let brothers = ['Nasir'];
brothers.push('Antor'); // ['Nasir', 'Antor']
brother.push('Apon', 'Uddin'); // ['Nasir', 'Antor', 'Apon', 'Uddin']

# shift() : Remove an item from the beginning of an array.

let brothers = ['Nasir', 'Antor', 'Apon'];
brothers.shift(); // ['Antor', 'Apon'];

# unshift() : Add items to the beginning of an array.

let brothers = ['Uddin'];
brothers.unshift('Apon'); // ['Apon', 'Uddin']
brothers.unshift('Nasir', 'Antor'); // ['Nasir', 'Antor', 'Apon', 'Uddin']
